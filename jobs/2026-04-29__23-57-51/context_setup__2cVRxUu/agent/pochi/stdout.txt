  getUser: publicProcedure.query(({ ctx }) => {

### `src/app/api/trpc/[trpc]/route.ts` — Next.js catch-all API route
```ts
import { fetchRequestHandler } from '@trpc/server/adapters/fetch';
import { type NextRequest } from 'next/server';
import { appRouter } from '@/trpc/router';
import { type Context } from '@/trpc/init';

function createContext({ req }: { req: NextRequest }): Context {
  const userId = req.headers.get('x-user-id') ?? 'anonymous';
  return { userId };
}

const handler = (req: NextRequest) =>
  fetchRequestHandler({ endpoint: '/api/trpc', req, router: appRouter, createContext });

export { handler as GET, handler as POST };
```
`createContext` reads `x-user-id` from the standard Web API `Request.headers` provided by `fetchRequestHandler` and falls back to `'anonymous'`.
