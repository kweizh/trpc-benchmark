export const t = initTRPC.create({
  errorFormatter({ shape, error }) {
    if (error.code === 'BAD_REQUEST' && error.cause instanceof ZodError) {
      return {
        ...shape,
        data: {
          ...shape.data,
          zodError: error.cause.flatten(),
        },
      };
    }
    return {
      ...shape,
      data: {
        ...shape.data,
        zodError: null,
      },
    };
  },
});
